159 void ServoSteuerung::SwitchInStart()
160 {
161  TCNT0 = 0; // measure timer startvalue = 0
162  SwitchTimerStart256(); // Prescaler 256 => counts max 4ms => max. servo dutycycle on 50Hz = 2.4ms = 180° (min 0.6ms = 0°)
163  if (m_driveToZeroState == TimerState::waiting) {
164  // PWM Timer start.If it is already running, driveToZeroCounter is set to 0, ServoPWM continues running
165  if (!(TCCR1B & (1 << CS11))) {
166  TCNT1 = 0;
167  PWMTimerStart8(); // set Prescaler (Timer is on)
168  }
169  TCNT2 = 0;
170  DtZTimerStart1024(); // start dTZ Timer with prescaler 1024 (ca. 61/s)
171  m_driveToZeroState = TimerState::counting;
172  PORTB |= (1 << PORTB2); //TODO: Testpin entfernen
173  }
174 }

263 void ServoSteuerung::StopPWM()
264 {

282 void ServoSteuerung::initPins()
283 {

285  ServoOpenPort &= ~((1 << SV12OpenPin) | (1 << SV34OpenPin) | (1 << SV56OpenPin)); // All Servos signal off
286
